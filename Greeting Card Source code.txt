    (function createBackgroundHearts(){

    function startSparkleShower(){
      const duration = 30000;
      const end = Date.now() + duration;
      const timer = setInterval(()=>{
        for(let i=0;i<6;i++){
          const sp = document.createElement('div');
          sp.className = 'falling';
          sp.textContent = 'üå∏';
          sp.style.left = Math.random()*window.innerWidth + 'px';
          sp.style.animationDuration = (2 + Math.random()*2) + 's';
          document.body.appendChild(sp);
          setTimeout(()=>sp.remove(), 4000);
        }
        if(Date.now()>end) clearInterval(timer);
      },180);
    }

    (function untouchableNo(){
      const no = document.getElementById('no');
      const yes = document.getElementById('yes');
      const controls = document.getElementById('controls');
      const bouquet = document.getElementById('bouquet');
      const letterLink = document.getElementById('letterLink');
      const overlay = document.getElementById('overlay');

      const pos = { x:0, y:0 };
      let target = { x:0, y:0 };
      let initial = { x:0, y:0 };
      const reactDistance = 140; 
      const moveDistance = 60;   
      const margin = 10;

      function placeInitialNo(){
        const controlsRect = controls.getBoundingClientRect();
        const yesRect = yes.getBoundingClientRect();
        initial.x = (yesRect.right - controlsRect.left) + 28;
        initial.y = (yesRect.top - controlsRect.top);
        pos.x = initial.x; pos.y = initial.y;
        target.x = initial.x; target.y = initial.y;
        no.style.left = pos.x + 'px';
        no.style.top = pos.y + 'px';
      }

      function animate(){
        pos.x += (target.x - pos.x) * 0.4;
        pos.y += (target.y - pos.y) * 0.4;
        no.style.left = pos.x + 'px';
        no.style.top = pos.y + 'px';
        requestAnimationFrame(animate);
      }

      function onPointerMove(clientX, clientY){
        const rect = no.getBoundingClientRect();
        const cx = rect.left + rect.width/2;
        const cy = rect.top + rect.height/2;
        const dx = cx - clientX;
        const dy = cy - clientY;
        const distance = Math.hypot(dx, dy);

        if(distance < reactDistance){
          let moveX = dx / distance * moveDistance;
          let moveY = dy / distance * moveDistance;
          target.x = clamp(initial.x + moveX, margin, window.innerWidth - rect.width - margin - controls.getBoundingClientRect().left);
          target.y = clamp(initial.y + moveY, margin, window.innerHeight - rect.height - margin - controls.getBoundingClientRect().top);
        } else {
          target.x = initial.x;
          target.y = initial.y;
        }
      }

      function clamp(v,a,b){ return Math.max(a, Math.min(b,v)); }

      window.addEventListener('mousemove', e=>requestAnimationFrame(()=>onPointerMove(e.clientX,e.clientY)), {passive:true});
      window.addEventListener('mousedown', e=>requestAnimationFrame(()=>onPointerMove(e.clientX,e.clientY)), {passive:true});
      window.addEventListener('resize', placeInitialNo);
      requestAnimationFrame(animate);

      yes.addEventListener('click', ()=>{
        bouquet.style.display='block';
        controls.style.display='none';
        letterLink.style.display='inline-block';
        document.getElementById('question').textContent='I knew you would say YES!üíñ';
        startSparkleShower();
      });

      letterLink.addEventListener('click', ()=>{
        overlay.style.display='flex';
        createOverlayHearts();
      });

      document.getElementById('closeOverlay').addEventListener('click', ()=>{
        overlay.style.display='none';
        document.querySelectorAll('#overlay .sparkle').forEach(s=>s.remove());
      });

      placeInitialNo();

      function createOverlayHearts(){
        const count = 70;
        for(let i=0;i<count;i++){
          const s = document.createElement('div');
          s.className = 'sparkle';
          s.textContent = '‚ù§';
          s.style.fontSize = 13 + Math.random()*16 + 'px';
          s.style.left = Math.random() * window.innerWidth + 'px';
          s.style.top  = Math.random() * window.innerHeight + 'px';
          s.style.animationDuration = (1 + Math.random()*2) + 's';
          overlay.appendChild(s);
        }
      }
    })();
